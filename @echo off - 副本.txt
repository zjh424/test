# VisualSVN Server Web Management Interface - 完整功能版
<Location /svn-admin/>
    DAV visualsvn
    SVNParentPath "E:/data/files/"
    SVNListParentPath on
    
    # 认证
    AuthName "VisualSVN Server Administration"
    AuthType Basic
    AuthBasicProvider file
    AuthUserFile "E:/data/files/htpasswd"
    
    # 授权（权限控制）
    AuthzVisualSVNMode Subversion
    AuthzVisualSVNSubversionReposRelativeAccessFile "VisualSVN-SvnAuthz.ini"
    AuthzVisualSVNSubversionGlobalAccessFile "E:/data/files/VisualSVN-GlobalSvnAuthz.ini"
    AuthzVisualSVNSubversionGroupsFile "E:/data/files/groups.conf"
    VisualSVNEnableListAuthz on
    
    # 管理功能增强
    VisualSVNEnableRepositoryCreation on
    VisualSVNEnableRepositoryDeletion on
    
    Require valid-user
</Location>
